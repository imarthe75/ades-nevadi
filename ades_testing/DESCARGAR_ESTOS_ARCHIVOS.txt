   $ scp config_ades_modules.json ubuntu@setag-arm-ashburn:/opt/ades/ades_testing/

3. LEER instrucciones:
   $ ssh ubuntu@setag-arm-ashburn
   $ cd /opt/ades/ades_testing
   $ cat INSTRUCCIONES_SERVIDOR_UBUNTU.md

4. EJECUTAR scripts:
   $ source venv/bin/activate
   $ source .env
   $ python 01_ades_explorer_portable.py
   $ python 02_claude_qa_analyzer_portable.py
   $ python 03_report_generator_portable.py

5. VER resultados:
   $ ls reports/
